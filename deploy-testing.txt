#!/bin/bash
# deploy-testing.sh — Deploy ps2-picker.py to the TESTING branch only
# Usage: ~/deploy-testing.sh <patch-file> "Commit message"
set -euo pipefail

LIVE=~/ps2-picker.py
REPO=~/ps2-picker-repo
RED='\033[0;31m'
GRN='\033[0;32m'
YEL='\033[1;33m'
RST='\033[0m'

die()  { echo -e "${RED}✗ $1${RST}" >&2; exit 1; }
ok()   { echo -e "${GRN}✓ $1${RST}"; }
info() { echo -e "${YEL}► $1${RST}"; }

if [[ $# -lt 2 ]]; then
    echo "Usage: ~/deploy-testing.sh <patch-file> \"Commit message\""
    exit 1
fi

SOURCE="$1"
MSG="$2"

[[ -f "$SOURCE" ]] || die "File not found: $SOURCE"

info "Syntax checking..."
python3 -m py_compile "$SOURCE" || die "Syntax error in patch file!"
ok "Syntax OK"

TIMESTAMP=$(date +%Y%m%d-%H%M%S)
if [[ -f "$LIVE" ]]; then
    BAK="${LIVE}.${TIMESTAMP}.bak"
    cp "$LIVE" "$BAK"
    ok "Backup → $BAK"
fi

info "Deploying to live..."
cp "$SOURCE" "$LIVE"
chmod +x "$LIVE"
ok "Updated $LIVE"

info "Pushing to testing branch..."
cd "$REPO"
git checkout testing 2>/dev/null || git checkout -b testing
cp "$LIVE" "$REPO/ps2-picker.py"
git add ps2-picker.py
git commit -m "$MSG"
git push origin testing
ok "Pushed to testing: $MSG"

echo ""
echo -e "${GRN}═══ Deployed to TESTING branch only ═══${RST}"
